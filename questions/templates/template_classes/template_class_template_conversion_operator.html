#pragma once

#include <iostream>


template <typename T>
class Base
{
public:
    Base() = default;

    template <typename U>
    operator Base<U>()
    {
        return Base<U>{};
    }

    void print();
private:
    T val;
};

template <typename T>
void Base<T>::print()
{
    std::cout << "val: " << val << '\n';
}

    auto baseT3 = static_cast<Base<std::complex<double>>>(baseDouble);

    baseT2.print();
    baseT3.print();
