class.h 

#pragma once

#include <iostream>


class Base
{
private:
    int private_x { 111 };
protected:
    int protected_x { 222 };
public:
    int public_x { 333 };

    virtual void func()
    {
        std::cout << "Base func()\n";
    }
};

class Derived : public Base
{
private:
    int private_x { 1 };
protected:
    int protected_x { 2 };
public:
    int public_x { 3 };

    void func()
    {
        std::cout << "Derived func()\n";
    }
};

main.cpp 

#include "class.h"
#include <iostream>


int main()
{
    Base* ptr1 = new Base();
    std::cout << ptr1->public_x << '\n';
    ptr1->func();

    Derived* ptr2 = new Derived();
    std::cout << ptr2->public_x << '\n';
    ptr2->func();

    Base* ptr3 = new Derived();
    std::cout << ptr3->public_x << '\n';
    ptr3->func(); 

    delete ptr3;
    delete ptr2;
    delete ptr1;

    return 0;
}
